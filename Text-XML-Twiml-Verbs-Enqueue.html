<Enqueue>

This example 

module Example where

import Control.Lens
import Data.Maybe (fromJust)
import Text.XML.Twiml

example
  = respond
  . (enqueue "support" <&> waitURL .~ (fromJust $ parseURL "wait-music.xml"))
  $ end
