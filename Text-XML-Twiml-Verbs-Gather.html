<Gather>

This example 

module Example where

import Control.Lens
import Text.XML.Twiml

example
  = respond
  . (gather <&> timeout     .~ 10
            <&> finishOnKey .~ KStar
      $ say "Please enter your pin number and then press star."
      $ end)
  $ end

produces the following TwiML response: 

<?xml version="1.0" encoding="UTF-8"?>

  </Gather>
</Response>
