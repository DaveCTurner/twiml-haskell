pause :: IsTwimlLike f Pause => PauseAttributes -> TwimlLike f Pause ()

Example:

example :: VoiceTwiml
example =
  response $ do
    say "I will pause 10 seconds starting now!" def
    pause $ def & duration .~ Just 10
    say "I just paused 10 seconds" def
    end
  where Twiml.Syntax{..} = def
>>> show example
