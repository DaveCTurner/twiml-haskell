This example 

module Example where

import Text.XML.Twiml

example
  = respond
  . sayWoman' French "Chapeau"
  $ end

produces the following TwiML response: 

<?xml version="1.0" encoding="UTF-8"?>
<Response>
  <Say voice="woman" language="fr">Chapeau</Say>
</Response>
